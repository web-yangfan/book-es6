模板字符串

let json = {
  name: '艾斯'
}

let age = 23

let gender = () => {
  return '男'
}

let str = `我的名字是：${json.name}，我的性别是:${age}，我的年纪是:${gender()}`;
console.log(str) // 我的名字是：艾斯，我的性别是:23，我的年纪是:男


includes()

返回布尔值，表示是否找到了参数字符串。
var s = 'Hello world';
s.includes('o'); // true

// 第二个参数，表示开始搜索的位置
s.includes('Hello', 6); // false


startsWith()

返回布尔值，表示参数字符串是否在源字符串的头部。
var s = 'Hello world';
s.startsWith('Hello'); // true

// 第二个参数，表示开始搜索的位置
s.startsWith('world', 6); // true


endsWith()

返回布尔值，表示参数字符串是否在源字符串的尾部
var s = 'Hello world';
s.endsWith('!'); // false

// 第二个参数，表示开始搜索的位置
s.endsWith('Hello', 5)； // true


repeat()

返回一个新字符串，表示将原字符串重复n次
'x'.repeat(3);  // xxx


padStart()，padEnd()

字符串补全长度的功能
'x'.padStart(4, 'ab') // 'abax'
'x'.padEnd(5, 'ab') // 'xabab'
